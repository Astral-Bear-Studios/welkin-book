term ::= graph | connection | ident | string | num
graph ::= graph_ident? '{' term '}'
graph_ident ::= ident | string
connection ::= term connector term
connector ::= edge | arrow
edge ::= '-' term '-'
left_arrow ::= '-' term '>'
right_arrow ::= '<' term '-'
string ::= '"' CHAR* '"' | ['] CHAR* [']
ident ::= CHAR*
num ::= NUMBER
