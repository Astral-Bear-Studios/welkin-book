terms       = term*
term        = graph | connections | member | unit
graph       = (ident | string)? '{' term '}'
connections = term (connector term)+
connector   = edge | left_arrow | right_arrow
edge        = '-' term '-'
left_arrow  = '-' term '>'
right_arrow = '<' term '-'
member      = unit '.' (unit '.')* unit
unit        = ident | num | string
ident       = CHAR*
num         = NUMBER
string      = STRING
